วิธีเชื่อม Google Sheets (ทำครั้งเดียว)
  1. เปิด Google Sheets → เมนู Extensions → Apps Script
  2. ลบโค้ดเดิมออก แล้ววางโค้ดนี้แทน:
    function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheets = ['Nampu','Earth','Pooh']; const result = []; sheets.forEach(sName => { const s = ss.getSheetByName(sName); if (!s) return; s.getDataRange().getValues().forEach((row,i) => { if (i===0) return; const num = row[0]?.toString().trim(); const bon = parseFloat(row[1])||0; const lang = parseFloat(row[2])||0; const toad = parseFloat(row[3])||0; const name = row[4]?.toString().trim(); if (!num||(!bon&&!lang&&!toad)||isNaN(parseInt(num))) return; result.push({sheet:sName[0],num,bon,lang,toad,name:name||'?',total:bon+lang+toad}); }); }); return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); }
  3. กด Deploy → New deployment
  4. Type: Web app → Execute as: Me → Who has access: Anyone
  5. กด Deploy → Copy URL ที่ได้
  6. วาง URL ในช่องด้านบน แล้วกด โหลด
